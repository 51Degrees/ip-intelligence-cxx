59 template<typename T>
60 VariableSizeCollection<T>::VariableSizeCollection(const std::vector<T> &values) {
61  uint32_t currentOffsetIndex = 0;
62  fiftyoneDegreesMemoryReader reader;
63  size_t dataLength = values.size() * sizeof(T);
64  state.count = (uint32_t) values.size();
65 
66  reader.length = (FileOffset)(dataLength + sizeof(uint32_t));
67  state.data = fiftyoneDegreesMalloc(reader.length);
68  *(int32_t*)state.data = (int32_t)dataLength;
69  state.offsets = (uint32_t*)fiftyoneDegreesMalloc(values.size() * sizeof(uint32_t));
70  reader.startByte = ((byte*)state.data);
71  reader.lastByte = reader.startByte + reader.length;
72  reader.current = reader.startByte + sizeof(uint32_t);
73 
74  for (size_t i = 0; i < values.size(); i++) {
75  T *element = (T*)reader.current;
76  memcpy((void *) element, (void *)&values[i], sizeof(T));
77  state.offsets[currentOffsetIndex] =
78  (uint32_t)(reader.current - (reader.startByte + sizeof(uint32_t)));
79  reader.current += sizeof(T);
80  currentOffsetIndex++;
81  }
82  assert(currentOffsetIndex == state.count);
83  assert(reader.lastByte == reader.current);
84  assert((byte*)state.data + sizeof(int32_t) + dataLength == reader.lastByte);
85  assert((byte*)state.data + reader.length == reader.lastByte);
86  reader.current = reader.startByte;
87  state.collection = fiftyoneDegreesCollectionCreateFromMemory(
88  &reader,
89  fiftyoneDegreesCollectionHeaderFromMemory(&reader, 0, false));
90  assert(state.collection->size == dataLength);
91 }
