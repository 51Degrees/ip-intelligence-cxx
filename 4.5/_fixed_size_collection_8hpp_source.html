60template<typename T>
61FixedSizeCollection<T>::FixedSizeCollection(const std::vector<T> &values) {
62 uint32_t currentOffsetIndex = 0;
63 fiftyoneDegreesMemoryReader reader;
64 size_t dataLength = values.size() * sizeof(T);
65 state.count = (uint32_t) values.size();
66
67 reader.length = (fiftyoneDegreesFileOffset)(dataLength + sizeof(uint32_t));
68 state.data = fiftyoneDegreesMalloc(reader.length);
69 *(int32_t*)state.data = (int32_t)dataLength;
70 state.offsets = (uint32_t*)fiftyoneDegreesMalloc(values.size() * sizeof(uint32_t));
71 reader.startByte = ((byte*)state.data);
72 reader.lastByte = reader.startByte + reader.length;
73 reader.current = reader.startByte + sizeof(uint32_t);
74
75 for (size_t i = 0; i < values.size(); i++) {
76 T *element = (T*)reader.current;
77 memcpy((void *) element, (void *) &values[i], sizeof(T));
78 state.offsets[currentOffsetIndex] =
79 (uint32_t)(reader.current - (reader.startByte + sizeof(uint32_t)));
80 reader.current += sizeof(T);
81 currentOffsetIndex++;
82 }
83 assert(currentOffsetIndex == state.count);
84 assert(reader.lastByte == reader.current);
85 assert((byte*)state.data + sizeof(int32_t) + dataLength == reader.lastByte);
86 assert((byte*)state.data + reader.length == reader.lastByte);
87 reader.current = reader.startByte;
88 state.collection = fiftyoneDegreesCollectionCreateFromMemory(
89 &reader,
90 fiftyoneDegreesCollectionHeaderFromMemory(&reader, sizeof(T), false));
91 assert(state.collection->size == dataLength);
92}
